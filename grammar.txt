denial :
	expressions

expressions :
	expression COMMA expressions |
	expression

expression:
	path_exp |
	path_exp COMPARE path_exp

path_exp:
	path_start path_cont

path_start:
	empty |
	ID |
	element |

path_cont:
	SLASH CALL bind |
	SLASH ATTRIB bind |
	SLASH UP bind path_cont |
	SLASH STAR bind path_cont |
	SLASH ELEMENT bind path_cont |
	DBLSLASH ELEMENT bind path_cont |
